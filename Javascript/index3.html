var resultado; var num1 = prompt("Indica un numero:"); var num2 = prompt("Indica otro numero:"); var operacion = prompt("Introduce: 1 para sumar, 2 para restar, 3 para multiplicar, 4 para dividir"); if (operacion == 1) resultado = parseInt(num1) + parseInt(num2); else if (operacion == 2) resultado = parseInt(num1) - parseInt(num2); else if (operacion == 3) resultado = parseInt(num1) * parseInt(num2); else if (operacion == 4) resultado = parseInt(num1) / parseInt(num2); alert(resultado);
